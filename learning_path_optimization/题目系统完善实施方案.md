#### 题目11：冒泡排序
```python
def bubble_sort(arr):
    """
    使用冒泡排序对数组进行升序排序
    
    Args:
        arr: 待排序的列表
    
    Returns:
        排序后的列表
    """
    n = len(arr)
    for i in range(n):
        # 标记是否发生交换
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        # 如果没有发生交换，说明已经有序
        if not swapped:
            break
    return arr

# 读取输入
n = int(input())
arr = list(map(int, input().split()))
result = bubble_sort(arr)
print(' '.join(map(str, result)))
```

#### 题目12：二分查找
```python
def binary_search(arr, target):
    """
    在有序数组中使用二分查找目标值
    
    Args:
        arr: 有序数组
        target: 目标值
    
    Returns:
        目标值的索引，如果不存在则返回-1
    """
    left, right = 0, len(arr) - 1
    
    while left <= right:
        mid = (left + right) // 2
        
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1

# 读取输入

```

import java.util.Scanner;

abstract class Animal {
    protected String name;
    
    public Animal(String name) {
        this.name = name;
    }
    
    public abstract void speak();
}

class Dog extends Animal {
    public Dog(String name) {
        super(name);
    }
    
    @Override
    public void speak() {
        System.out.println(name + " says: 汪汪汪");
    }
}

class Cat extends Animal {
    public Cat(String name) {
        super(name);
    }
    
    @Override
    public void speak() {

    }
}

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String type = scanner.nextLine();
        String name = scanner.nextLine();
        
        Animal animal;
        if (type.equals("Dog")) {
            animal = new Dog(name);
        } else {
            animal = new Cat(name);
        }
        animal.speak();
    }
}
```

class Animal(ABC):

**题目描述**：
设计一个Shape接口，包含计算面积的方法。实现Circle和Rectangle两个类。

**输入格式**：
第一行：图形类型（"Circle"或"Rectangle"）
如果是Circle，第二行：半径r
如果是Rectangle，第二行：宽width 高height

**输出格式**：
图形的面积，保留2位小数

**输入样例**：
```
Circle
5
```

**输出样例**：
```

**Java参考答案**：
```java
import java.util.Scanner;

interface Shape {
    double getArea();
}

class Circle implements Shape {
    private double radius;
    
    public Circle(double radius) {
        this.radius = radius;
    }
    
    @Override
    public double getArea() {
        return Math.PI * radius * radius;
    }
}

class Rectangle implements Shape {
    private double width;
    private double height;
    
    public Rectangle(double width, double height) {
        this.width = width;
        this.height = height;
    }
    
    @Override
    public double getArea() {
        return width * height;
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String type = scanner.nextLine();
        
        Shape shape;
        if (type.equals("Circle")) {
            double radius = scanner.nextDouble();
            shape = new Circle(radius);
        } else {
            double width = scanner.nextDouble();
            double height = scanner.nextDouble();
            shape = new Rectangle(width, height);
        }
        
        System.out.printf("%.2f", shape.getArea());
    }
}
```

**Python参考答案**：
```python
from abc import ABC, abstractmethod
import math

class Shape(ABC):
    @abstractmethod
    def get_area(self):
        pass

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def get_area(self):
        return math.pi * self.radius ** 2

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def get_area(self):
        return self.width * self.height

shape_type = input()

if shape_type == "Circle":
    radius = float(input())
    shape = Circle(radius)
else:
    width, height = map(float, input().split())
    shape = Rectangle(width, height)

print(f"{shape.get_area():.2f}")
```

#### 新题目15：学生管理系统（控制台应用）
**题目描述**：
实现一个简单的学生管理系统，支持添加学生、删除学生、查询学生、显示所有学生功能。

**输入格式**：
多行输入，每行一个操作：
- "add name score"：添加学生
- "delete name"：删除学生
- "query name"：查询学生
- "list"：显示所有学生
- "exit"：退出

**输出格式**：
根据操作输出相应结果

**输入样例**：
```
add 张三 90
add 李四 85
list
query 张三
delete 张三
list
exit
```

**输出样例**：

添加成功
添加成功

**Java参考答案**：
```java
import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Map<String, Integer> students = new HashMap<>();
        
        while (scanner.hasNextLine()) {
            String line = scanner.nextLine();
            String[] parts = line.split(" ");
            String op = parts[0];
            
            if (op.equals("exit")) {
                break;
            } else if (op.equals("add")) {
                String name = parts[1];
                int score = Integer.parseInt(parts[2]);
                students.put(name, score);
                System.out.println("添加成功");
            } else if (op.equals("delete")) {
                String name = parts[1];
                students.remove(name);
                System.out.println("删除成功");
            } else if (op.equals("query")) {
                String name = parts[1];
                System.out.println(name + ": " + students.get(name));
            } else if (op.equals("list")) {
                for (Map.Entry<String, Integer> entry : students.entrySet()) {
                    System.out.println(entry.getKey() + ": " + entry.getValue());
                }
            }
        }
    }
}
```
