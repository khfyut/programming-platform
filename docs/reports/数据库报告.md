```sql
SELECT 'problem' AS table_name, COUNT(*) AS row_count FROM problem
UNION ALL
SELECT 'learning_path', COUNT(*) FROM learning_path
UNION ALL
SELECT 'path_chapter', COUNT(*) FROM path_chapter
UNION ALL
SELECT 'path_level', COUNT(*) FROM path_level
UNION ALL
SELECT 'path_level_problem', COUNT(*) FROM path_level_problem
UNION ALL
SELECT 'knowledge_point', COUNT(*) FROM knowledge_point
UNION ALL
SELECT 'test_case', COUNT(*) FROM test_case
UNION ALL
SELECT 'user', COUNT(*) FROM user
UNION ALL
SELECT 'submit', COUNT(*) FROM submit
UNION ALL
SELECT 'wrong_book_item', COUNT(*) FROM wrong_book_item
UNION ALL
SELECT 'ai_session', COUNT(*) FROM ai_session
UNION ALL
SELECT 'community_post', COUNT(*) FROM community_post;

SELECT COUNT(*) AS orphan_test_cases
FROM test_case tc
LEFT JOIN problem p ON p.id = tc.problem_id
WHERE p.id IS NULL;

DELETE tc
FROM test_case tc
LEFT JOIN problem p ON p.id = tc.problem_id
WHERE p.id IS NULL;
```
